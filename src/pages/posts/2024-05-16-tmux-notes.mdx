prefix: C + b

- reload tmux conf: [prefix]: source ~/.tmux.conf
- tmux ls
- tmux new [-s][sessionname]

## session

- session list: [prefix] + s
- delete(kill) session: show session list, :kill-session

## window

- create window: [prefix] + c
- rename current window: [prefix] + ,
- switch to next window: [prefix] + n
- switch to previous window: [prefix] + p
- list windows: [prefix] + w
- delete(kill) window: show window list, :kill-window

- split window -h: [prefix] + %
- split window -v: [prefix] + "

## pane

- switch pane: [prefix] + arrows
